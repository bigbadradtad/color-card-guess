  <script>

    // Weighted / streaky color generator

    function pickWeighted(weights) {

      const entries = Object.entries(weights);

      let total = entries.reduce((a, [,w]) => a + w, 0);

      let r = Math.random() * total;

      for (let [k, w] of entries) {

        if (r < w) return k;

        r -= w;

      }

    }


    const Generator = (() => {

      let prev = null;

      let streak = 0;


      return {

        next() {

          if (streak > 0) {

            streak--;

            return prev;

          }

          if (Math.random() < 0.15) {

            prev = pickWeighted({red: 0.4, blue: 0.3, yellow: 0.3});

            streak = Math.floor(Math.random() * 2) + 1;

            return prev;

          }

          const weights = { red: 0.36, blue: 0.32, yellow: 0.32 };

          if (prev) weights[prev] += 0.15;

          prev = pickWeighted(weights);

          return prev;

        }

      }

    })();


    const card = document.getElementById('card');

    const feedback = document.getElementById('feedback');

    const options = document.querySelectorAll('input[name="guess"]');


    let actualColor = null;

    let revealed = false;


    card.addEventListener('click', () => {

      const guess = [...options].find(o => o.checked)?.value;

      if (!guess) {

        feedback.textContent = "Pick a color first!";

        return;

      }

      if (revealed) return;


      actualColor = Generator.next();

      card.textContent = actualColor === "red" ? "🔴" :

                         actualColor === "blue" ? "🔵" : "🟡";


      if (guess === actualColor) {

        feedback.textContent = "✅ Correct!";

      } else {

        feedback.textContent = `❌ Wrong! It was ${actualColor.toUpperCase()}.`;

      }

      revealed = true;

    });


    card.addEventListener('dblclick', () => {

      // Reset for next round

      card.textContent = "?";

      feedback.textContent = "";

      revealed = false;

    });

  </script>
